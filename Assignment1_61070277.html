def add_data2csv(file): # file 
    for i in range(25): # 25 
        page=i+1
        url ='https://api.rawg.io/api/games?page='+str(page) # api  
        response = requests.get(url)
        restext=response.text
        js = pd.io.json.loads(restext)
        with open(file, "a", newline="") as write_file: #
            f = csv.writer(write_file)
            for i in js['results']: #loop
                f.writerow((i['name'],i['released'],i['rating']))
            write_file.close()
